"""Deletes the columns, where marital status is not single nor married"""
data = data[data["Marital status"].isin([1, 2])]

"""Changes column to Single/Married (text) instead of 1/2 (num)"""
data["Marital status"] = data["Marital status"].apply(lambda x: "Single" if x == 1 else "Married")

"""Changes the number to the name of the course"""
data["Course"] = data["Course"].apply(lambda x: get_course(x))

"""Changes the number to the belonging nationality"""
data["Nationality"] = data["Nationality"].apply(lambda x: get_country(x))

"""Changes the number to the belonging education level of the parents"""
data["Mother\'s qualification"] = data["Mother\'s qualification"].apply(lambda x: get_parents_education_level(x))
data["Father\'s qualification"] = data["Father\'s qualification"].apply(lambda x: get_parents_education_level(x))

"""Changes the number to if the person was displaced or not"""
data["Displaced"] = data["Displaced"].apply(lambda x: "Displaced" if x == 1 else "Not displaced")

"""Changes the number to the belonging gender"""
data["Gender"] = data["Gender"].apply(lambda x: "Male" if x == 1 else "Female")

"""Changes the number to if they get a scholarship or not"""
data["Scholarship holder"] = data["Scholarship holder"].apply(
        lambda x: "Scholarship" if x == 1 else "No scholarship")

dataMan = data[data["Target"] != "Enrolled"]
contingency_table = pd.crosstab(dataMan['Scholarship holder'], dataMan['Target'], margins=True)
print(contingency_table,'\n')

# Chi-Quadrat-Test
chi2, p_value, dof, expected = stats.chi2_contingency(contingency_table.iloc[:-1, :-1])

# Prints the results
print(f"Chi2-Value: {chi2}")
print(f"p-Value: {p_value}")
print()

if p_value < 0.05:
        print("There is a significant correlation between scholarship status and target status.")
        print("Expected values:")
        print(expected)
else:
        print("There is no significant correlation between scholarship status and target status.")
